import random


def getNum():
  myList = [(1, .5), (2, .4), (3, .1)]
  myProb =   random.random()
  # print(myProb, end="\t")
  total = 0
  for num, prob in myList:
    total += prob
    if myProb < total:
      return num

counts = {1: 0, 2: 0, 3: 0}
for i in range(100000):
  counts[getNum()] += 1
print(counts)

// Protects against invalid RequestType values
Model() {
  int index = config.getInt("RequestType");
  if (index > 2) {
    System.out.error(String.format("Error. RequestType in %s cannot be greater than %d. Value: %d", modelName, RequestType.values().length, index);
    requestType = RequestType.GET;
  } else if (index < 0) {
    System.out.error(String.format("Error. RequestType in %s cannot be less than 0. Value: %d", modelName, index);
    requestType = RequestType.GET;
  } else {
    requestType = RequestType.values()[index];
  }
}
